pipeline {
    agent none
    stages {
        stage('Pre-setup'){
            steps {
                echo 'Hello world'
            }
        }
        stage('Parallel tests') {
            parallel {
                stage('Test On Windows') {
                    agent {
                        label "windows"
                    }
                    stages {
                        stage('Windows-1'){
                            steps{
                            sh "echo 'Windows 1'"
                            sh "error"
                            }
                        }
                        stage('Windows-2'){
                            steps{
                            sh "echo 'Windows 2'"
                            sh "error"
                            }
                        }
                        stage('Windows-3'){
                            steps{
                            sh "echo 'Windows 3'"
                            sh "error"
                            }
                        }
                    }
                    post {
                        always {
                            sh "echo 'done'"
                            sh "error"
                        }
                    }
                }
                stage('Test On Linux') {
                    agent {
                        label "master"
                    }
                    steps {
                        sh "echo 'Linux'"
                        sh "exit 1"
                    }
                    post {
                        always {
                            sh "echo 'done'"
                            sh "exit 1"
                        }
                    }
                }
            }
        }
    }
}
